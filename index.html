import numpy as np
def main():
    number_file = open("numbers.txt", "r")
    count = 0
    total = 0
	
    
    for line in number_file:
        count += 1	
        number = int(line)
        total += number
    average = total / count
	
    
    print(("Sum:"),(total))
    print(("Count:"),(count))
    print(("Average:"),(average))
    x = np.loadtxt("numbers.txt")
    print("Max:", np.amax(x))
    print("Min:", np.amin(x)) 
    Range = np.amax(x) - np.amin(x) 
    print("Range:", Range)        
	
    
main()
